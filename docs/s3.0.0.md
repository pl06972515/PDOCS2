// 当前工作目录向上递归查找 [ 指定文件 ]
def find_file(filename: str, start_dir: str = None) -> Tuple[bool, Path]:

	current_dir = Path(start_dir or Path.cwd()).resolve()
	while True:
		_target = current_dir / filename
		if _target.is_file():
			return True, _target

		parent_dir = current_dir.parent
		if parent_dir == current_dir:  // 已经到达系统根目录
			break
		current_dir = parent_dir

	return False, None
