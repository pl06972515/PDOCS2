// [ 获取 ] 工作目录
def get_base_path():
    // [ 判断 ] 是否为打包后的程序运行
    if getattr(sys, 'frozen', False):
        base_path = sys._MEIPASS  // 打包后程序的临时目录
    else:
        base_path = os.getcwd()   // 源代码运行时的目录
    return base_path
