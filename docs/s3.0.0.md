/// <summary>
/// - [ 获取 ] 工作目录 
///   sys.frozen: 是一个由某些打包工具（如 PyInstaller、cx_Freeze 等）[ 在生成可执行文件时 - 添加的属性 ]
//                当程序是通过这些工具打包为独立可执行文件运行的 则 sys.frozen 会被设置为 True 否则 该属性不存在
/// 
/// </summary>
def get_base_path():
    if getattr(sys, 'frozen', False):
       base_path = sys._MEIPASS 
    else:
       base_path = os.getcwd()
    return base_path
