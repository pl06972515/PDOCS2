```csharp
from snowflake import SnowflakeGenerator
import hashlib
import struct

def get_worker_id_from_app_name(app_name: str) -> int:
	hash_bytes = hashlib.sha256(app_name.encode('utf-8')).digest()
	value = struct.unpack('<H', hash_bytes[ :2 ])[ 0 ]
	return value % 64
        
        
```

```csharp
# 使用示例
app_name = "PM2.WEBService.09"
worker_id = get_worker_id_from_app_name(app_name)
print(f"App Name: {app_name}, Worker ID: {worker_id}")

# [ 雪花算法 ] 创建生成器
generator = SnowflakeGenerator(worker_id)
snowflake_id = next(generator)
print(snowflake_id)
    

```
