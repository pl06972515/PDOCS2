#### **[ datetime ]日期时间**

!>`<class 'datetime.datetime'> : from datetime import datetime`

```csharp
{
    
      #region [ 常用函数 ]
         
      /// <summary>
      /// - [ 当前日期 + 时间 ]
      ///   d = datetime.now()  -- 2025-03-20 14:56:44.599159 ( <class 'datetime.datetime'> )
      ///   print(f"年: { d.year }, "
      ///         f"月: { d.month }, "
      ///         f"日: { d.day }, "
      ///         f"时: { d.hour }, "
      ///         f"分: { d.minute }, "
      ///         f"秒: { d.second }")
      ///         
      /// </summary>
      datetime.now() 
      datetime.utcnow()
          
      /// <summary>
      /// - [ 日期明细 ] <class 'time.struct_time'>
      ///   d = datetime.now()
      ///   st = d.timetuple()
      ///   print(f"年: { st.tm_year }, "
      ///         f"月: { st.tm_mon }, "
      ///         f"日: { st.tm_mday }, "
      ///         f"时: { st.tm_hour }, "
      ///         f"分: { st.tm_min }, "
      ///         f"秒: { st.tm_sec }"
      ///         f"星期几: { st.tm_wday }"
      ///         f"一年中第几天: { st.tm_yday }")
      ///         
      /// </summary>
      datetime.timetuple() 
      datetime.utctimetuple()
          
      datetime.weekday()     // [ 星期几 ] 星期一为 0, 星期日为 6
      datetime.isoweekday()  // [ 星期几 ] 星期一为 1, 星期日为 7
          
      /// <summary>
      /// - d = datetime.now() -- 2025-03-20 15:04:13.811642
      ///   x = v.date()       -- 2025-03-20
      ///   y = v.time()       -- 15:04:13.811642
      ///         
      /// </summary>
      datetime.date()  // [ 获取日期部分 ] <class 'datetime.date'>
      datetime.time()  // [ 获取时间部分 ] <class 'datetime.time'>
         
      // [ 创建日期对象 ] d = datetime(2019, 1, 1, 0, 0, 0, tzinfo=None)
      datetime(year, month, day, hour, minute, second, tzinfo)
          
      /// <summary>
      /// - [ 格式化 ] <class 'datetime.datetime'> → <class 'str'>
      ///   d1 = datetime.now()
      ///   s = d1.strftime("%Y-%m-%d %H:%M:%S")
      ///
      /// - [ 格式化 ] <class 'str'> → <class 'datetime.datetime'>
      ///   d2 = datetime.strptime(s, "%Y-%m-%d %H:%M:%S")
      ///         
      /// </summary>
      datetime.strptime(format)
      datetime.strptime(date_string, format)
          
      #region [ 日期操作 - timedelta ] from datetime import datetime, timedelta
       
      /// <summary>
      /// - EG：指定日期 +1天 +2小时 ( <class 'datetime.datetime'> )
      ///   d1 = datetime.now()
      ///   d2 = d1 + timedelta(days=1, hours=2)  
      ///         
      /// </summary>
      timedelta(days=0, seconds=0, microseconds=0, milliseconds=0, minutes=0, hours=0, weeks=0)
          
      /// <summary>
      //  - [ 日期差 ]
      ///   d1 = datetime.strptime("2023-10-03 11:00:00", "%Y-%m-%d %H:%M:%S")
      ///   d2 = datetime.strptime("2023-10-03 10:00:00", "%Y-%m-%d %H:%M:%S")
      ///   td = d1 - d2  -- ( <class 'datetime.timedelta'> )
      ///   ms = td.seconds
      ///   
      /// </summary>
      timedelta.days
      timedelta.seconds
      timedelta.microseconds
          
}


```



#### **[ datetime.date ]日期 **

!>`<class 'datetime.date'> : from datetime import date`

```csharp
{
      
       #region [ 常用函数 ]
           
       /// <summary>
       /// - [ 当前日期]
       ///   d = date.today()  -- 2025-03-20 ( <class 'datetime.date'> )
       ///   print(f"年: { d.year }, "
       ///         f"月: { d.month }, "
       ///         f"日: { d.day }, ")
       ///         
       /// </summary>
       date.today()
       
       /// <summary>
       /// - [ 日期明细 ] <class 'time.struct_time'>
       ///   d = date.today()
       ///   st = d.timetuple()
       ///   print(f"年: { st.tm_year }, "
       ///         f"月: { st.tm_mon }, "
       ///         f"日: { st.tm_mday }, "
       ///         f"星期几: { st.tm_wday }"
       ///         f"一年中第几天: { st.tm_yday }")
       ///         
       /// </summary>
       date.timetuple() 

       date.weekday()     // [ 星期几 ] 星期一为 0, 星期日为 6
       date.isoweekday()  // [ 星期几 ] 星期一为 1, 星期日为 7
       
       /// <summary>
       /// - [ 格式化 ] <class 'datetime.date'> → <class 'str'>
       ///   d1 = datetime.today()
       ///   s = d1.strftime("%Y-%m-%d")
       ///
       /// </summary>
       date.strptime(format)
       
       #region [ 日期操作 - timedelta ] from datetime import date, timedelta
       
       /// <summary>
       /// - EG： 指定日期 +1天 +2小时 ( <class 'datetime.date'> )
       ///    d1 = date.today()
       ///    d2 = d1 + timedelta(days=1, hours=2)  
       ///         
       /// </summary>
       timedelta(days=0, seconds=0, microseconds=0, milliseconds=0, minutes=0, hours=0, weeks=0)
          
       timedelta.days
       timedelta.seconds
       timedelta.microseconds
      
}



```



#### **[ ☢ timestamp ]时间戳**

```csharp
// [ 时间戳 ] <class 'datetime.datetime'>       
d1 = datetime.now()
t = d1.timestamp()             // 1742468663.913721 ( <class 'float'> )
d2 = datetime.fromtimestamp(t) // 2025-03-20 19:04:23.913721 ( <class 'datetime.datetime'> )

// [ 时间戳 ] <class 'datetime.date'>
d1 = datetime.now()
t = d1.timestamp()         // 1742468663.913721 ( <class 'float'> )
d2 = date.fromtimestamp(t) // 2025-03-20 ( <class 'datetime.date'> )
    
    
```
