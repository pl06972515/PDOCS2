>[!WARNING|style: flat|label: 简要说明 ]
>
>- 数值类型包含：
>
>   整数(`int`)
>
>   浮点数(`float`)
>
> 
>
>```csharp
>x = 10    // type: <class 'int'>
>x = 13.58 // type: <class 'float'>
>
>    
>    ```
>
>
>
><br/>

```csharp
{
    
     #region [ 内置 ] 常用函数
         
     int(x)   // [ 类型转换 ] 将`x`转换为一个整数
     float(x) // [ 类型转换 ] 将`x`转换为一个浮点数

     #region [ 内置 ] 数学函数
         
     abs(x)      // 绝对值
     max(*args)  // 最大值 x = max( 1, 3, 5, 7, 9 )
     min(*args)  // 最小值
     
     round(x, n) // 四舍五入( 保留 n 位小数 )

     #region [ import math ] 数学函数
         
     math.sqrt(x)  // 平方根
     math.pi       // 圆周率 π 的值。
     math.e        // 自然对数的底 e 的值。
        
}


```
