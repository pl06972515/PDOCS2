```shell
# range(stop): 生成从 0 到 stop-1 的整数序列
# [ 0, 1, 2, 3, 4 ]
for x in range(5):
    print(x)
        
# range(start, stop): 生成从 start 到 stop-1 的整数序列
# [ 5, 6, 7, 8, 9 ]
for x in range(5, 10):
    print(x)
        
# range(start, stop, step): 成从 start 到 stop-1 [ 步长为 step 的整数序列 ]
# [ 0, 2, 4, 6, 8 ]
for x in range(0, 10, 2):
    print(x)
        
        
```
