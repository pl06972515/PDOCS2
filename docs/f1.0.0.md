#### **os**

```csharp
import os
    
{
    
      #region [ 常用方法 ]
      
      // [ 工作目录 ]
      os.getcwd() -> <class 'str'>
      // [ 目录分割 ] \\
      os.path.sep -> <class 'str'>
          
      // [ 检测 ] 路径是否存在 ( 目录 | 文件 )
      os.path.exists(path)
      
      // [ 获取 ] 环境变量值
      os.getenv(key, [default=None]) 
      // [ 设置 ] 环境变量
      os.environ[key] = value
          
      // [ 执行系统命令 ] 并返回状态码
      os.system(command)
          
      #region [ 目录相关 ]
          
      // [ 检测 ] 路径是否为目录
      os.path.isdir(path)
          
      // [ 获取 ] 目录下文件 + 目录 ( 非递归 )
      os.listdir(path) -> <class 'list'>
          
      // [ 创建 ] 指定的目录 ( 单个目录：如果文件夹已经存在 - 则异常 )
      os.mkdir(dirPath)
      // [ 创建 ] 指定的目录 ( 递归目录：如果文件夹已经存在 - 则异常 )
      os.makedirs(dirPath)
          
      // [ 删除 ] 指定的目录 ( 单个目录：如果文件夹非空 - 则异常 )
      os.mkdir(dirPath)
      // [ 删除 ] 指定的目录 ( 递归目录：如果文件夹非空 - 则异常 )
      os.removedirs(dirPath)
      // [ 重命名 ] 
      os.rename(src, dst) 
          
      // [ 获取 ] 目录最后修改时间 ( 时间戳 )
      os.path.getmtime(path) 
          
      #region [ 文件相关 ]
          
      // [ 检测 ] 路径是否为文件
      os.path.isfile(filePath)
          
      /// <summary>
      /// - [ 获取分离 ] 文件名
      ///   v1 = os.path.basename(r"C:\Desktop\publish\1.txt") # <class 'str'> : 1.txt
      ///   v2 = os.path.splitext(r"C:\Desktop\publish\1.txt") # <class 'tuple'> : ('C:\\Desktop\\publish\\1', '.txt')
      ///
      /// </summary>
      os.path.basename(filePath)
      os.path.splitext(filePath)
          
      // [ 获取 ] 文件的大小
      os.path.getsize(filePath)
      // [ 获取 ] 文件最后修改时间 ( 时间戳 )
      os.path.getmtime(path) 
          
      // [ 删除 ] 指定的文件
      os.remove(filePath)
      // [ 重命名 ] 
      os.rename(src, dst) 
          
          
      #region [ 路径相关 ]
       
      // [ 路径拼接 ]    
      os.path.join(path, *paths)
      /// <summary>
      /// - [ 路径拆分 ]
      ///   v1 = os.path.split(r"C:\Desktop\publish")       -- <class 'tuple'> : ('C:\\Desktop', 'publish')
      ///   v2 = os.path.split(r"C:\Desktop\publish\1.txt") -- <class 'tuple'> : ('C:\\Desktop\\publish', '1.txt')
      ///
      /// </summary>
      os.path.split(path)
          
       /// <summary>
      /// - [ 获取 ] 绝对路径
      ///   v1 = os.path.abspath(r"C:\Desktop\publish")       -- <class 'str'> : C:\Desktop\publish
      ///   v2 = os.path.abspath(r"C:\Desktop\publish\1.txt") -- <class 'str'> : C:\Desktop\publish
      ///
      /// </summary>
      os.dirname(dirPath)
          
}


```



#### **[ ☢ 路径操作 ] pathlib**

!> <span style='color:red'>提供了一种面向对象的方式来操作路径，使得路径操作更加直观和易于理解</span>

```csharp
from pathlib import Path

{
     
      #region [ 常用方法 ]
          
      // [ 当前工作目录 ]
      Path.cwd() -> <class 'pathlib.WindowsPath'> 或 <class 'pathlib.PosixPath'>
      
      // [ 创建 ] Path 对象 ( 对象既可以是一个文件对象，也可以是一个文件夹对象 )
      Path(pathStr) -> <class 'pathlib.WindowsPath'> 或 <class 'pathlib.PosixPath'>
          
      p.exists()  // [ 检查 ] 路径是否存在
      p.is_dir()  // [ 检查 ] 路径是否为目录
      p.is_file() // [ 检查 ] 路径是否为文件
          
      #region [ 文件目录 ]
          
      /// <summary>
      /// - [ 遍历 ] 非递归
      ///   p = Path(f'D:\\771')
      ///   for i in p.iterdir():
      ///   print(i)
      ///
      /// </summary>
      p.iterdir()
      
      /// <summary>
      /// - [ 创建目录 ]
      ///   parents=True: 递归创建
      ///   exist_ok=True: 目录存在忽略异常
      ///
      /// </summary>
      p.mkdir(parents=True, exist_ok=True) 
          
      p.rmdir()                     // 删除 [ 空目录 ]
      p.unlink(missing_ok=False)    // 删除文件 ( missing_ok=True: 文件不存在 - 则忽略异常  )
      p.rename(dst)                 // [ 重命名 ] 文件或目录
          
      p.glob(pattern)               // [ 查找文件 ] 通配符模式
      
      #region [ 路径相关 ]
          
      // - [ 路径拼接 ] 通过 / 斜杠拼接
      //   p = Path('C:/Users/Administrator/Desktop/python三剑客')
      //   p1 = p/'pathlib库'
          
      p.parent  // [ 获取 ] 父目录
      p.name    // [ 获取 ] 文件名或目录名
      p.stem    // [ 获取 ] 文件名（不包括扩展名）
      p.suffix  // [ 获取 ] 文件扩展名
          
      p.resolve(strict=False) // 返回路径的绝对路径
          
}


```



#### **[ 复制移动 ]shutil**

!> 文件和目录的复制、移动、删除等高级操作

```csharp
import shutil
    
{
      // [ 拷贝文件 ] src表示源文件，dst表示目标文件夹
      shutil.copy(src, dst)
      // [ 拷贝目录 ] 只能复制到一个空文件夹 ( 否则异常 )
      shutil.copytree(src, dst)
          
      // [ 移动路径 ] 文件 或 目录 ( 目标文件夹不存在时 - 异常 )
      shutil.move(src, dst)
          
      // [ 删除目录 ] 递归彻底删除文件夹 ( 含非空目录 )
      shutil.move(src)
          
}


```



#### **[ 查找文件 ]glob**

>[!WARNING|style: flat|label: 简要说明 ]
>
>- 根据通配符模式 [ 查找文件 ]
>
>  \-`*`匹配任意数量的字符（不包括路径分隔符）
>
>  \-`?`匹配单个字符
>
>  \-`[seq]`匹配`seq`中的任意一个字符
>
>  \-`[!seq]`匹配不在`seq`中的任意一个字符
>
><br/>

```csharp
import glob
      
 {
      
       /// <summary>
       /// - [ 文件查找 ] 
       ///   recursive = True : 递归查找目录 ( 配合路径 ** )
       ///     
       ///   - [ 示例 ] 查找 771 + 子目录下所有 jpg 文件
       ///   path = r"D:\771\**\*.jpg"
       ///   vs = glob.glob(path, recursive = True)
       ///
       /// </summary>
       glob.glob(pattern, *, recursive=False) -> <class'list'>
       
       /// <summary>
       /// - [ 文件查找 ] 
       ///   for file in glob.iglob('**/*.py', recursive=True):
       ///   print(file)
       ///
       /// </summary>
       glob.iglob(pattern, *, recursive=False) -> iterable
        
 }


```
