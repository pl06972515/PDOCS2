```shell
[root@SYL ~]# wget https://www.python.org/ftp/python/3.12.6/Python-3.12.6.tgz

[root@SYL ~]# cd Python-3.12.6;
[root@SYL Python-3.12.6]# ./configure --prefix=/usr/local/Python-3.12.6; # [ 配置 ]编译后安装目录
```
