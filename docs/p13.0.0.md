#### **[ JSON →  Python ]**

```shell
import json

jsonStr = """
     {
         "Q1": "人员信息",
         "Q2": 2,

         "City": [ "北京", "上海", "广州" ],
         "Items": [
              { "Id": 1, "Name": "张三" },
              { "Id": 2, "Name": "李四" }
          ]
     }
     """

# <class 'dict'>
pJson = json.loads(jsonStr)

print(f"Q1: {pJson["Q1"]}")
print(f"Q2: {pJson["Q2"]}")

# <class 'list'>
for city in pJson["City"]:
    print(city)

# <class 'list'>
for item in pJson["Items"]:
    #<class 'dict'>
    for k,v in item.items():
        print(f"ID: {k}, Name: {v}")
            
            
```

```shell
# [ 文件读取 ]
with open('data.json', 'r', encoding= "UTF-8") as file:
    pJson = json.load(file)
    
        
```



#### **[ Python → JSON ]**

```shell
import json

# <class 'dict'>
pJson = {
    "Q1": "人员信息",
    "Q2": 2,
    "City": [ "北京", "上海", "广州" ],
    "Items": [
        { "Id": 1, "Name": "张三" },
        { "Id": 2, "Name": "李四" }
    ]
}

# <class 'str'>
jsonStr = json.dumps(pJson, ensure_ascii=False)


```

```shell
# [ 文件写入 ]
with open('data.json', 'w') as file:
     json.dump(pJson, file, ensure_ascii=False, encoding= "UTF-8")

    
```
