#### **[ 基本操作  ]**

```csharp
# [ 隐式包装 ] 携程对象
async def M1():
    return None  // [ 类似 ] Task.CompletedTask

async def M2():
    return 100   // [ 类似 ] Task.FromResult<int>(100)


```

```csharp
async def M3(delay: int) -> int:
	await asyncio.sleep(delay)
	return delay

# 1
if __name__ == "__main__":
   v = asyncio.run(M3(1))
   print(v)
        
        
```



#### **[ 异步迭代 ]**

```csharp
# 等待所有任务完成 ( 按提交顺序返回 )
async def MS() -> List[int]:
    async with asyncio.TaskGroup() as tg:
        tasks = [tg.create_task(M3(i)) for i in [1, 2, 3, 4, 5]]

    # 退出 TaskGroup 后 【 所有任务已完成 】
    res = [ x.result() for x in tasks ]
    return res

--------------------------------------------------------------
            
async def main():
	for x in await MS():
		print(x)

asyncio.run(main())
            
            
```

```csharp
# [ 非等待所有任务 ] 1,... 2,... 3,...
async def MS() -> AsyncIterator[int]:
    async with asyncio.TaskGroup() as tg:
        tasks = [tg.create_task(M3(i)) for i in [1, 2, 3, 4, 5]]
        for task in tasks:
            yield await task

--------------------------------------------------------------
                
async def main():
	# 在 Python 中 async for : 用于异步迭代
	async for x in MS():
		print(x)

asyncio.run(main())
            
            
```
